const SHEET_NAME = "Pagos_Cirugia"; function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(["Fecha","Paciente","Monto_Cirujano","Pago_Cirujano", "Anest_Nombre","Anest_Monto","Anest_Pago","Anest_Status", "Ayud1_Nombre","Ayud1_Monto","Ayud1_Pago","Ayud1_Status", "Ayud2_Nombre","Ayud2_Monto","Ayud2_Pago","Ayud2_Status"]); } sheet.appendRow(data.row); return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return ContentService.createTextOutput(JSON.stringify({ok:true,rows:[]})).setMimeType(ContentService.MimeType.JSON); const rows = sheet.getDataRange().getValues(); return ContentService.createTextOutput(JSON.stringify({ok:true,rows:rows})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})) .setMimeType(ContentService.MimeType.JSON); } }
